; Программа целочисленной арифметики
a 0		; Начинаем с нулевой ячейки памяти
i 100		; Записываем целое число 100 в ячейки с 0 по 3
i 200		; Записываем целое число 200 в ячейки с 4 по 7
i 5		; Записываем целое число 5 в ячейки с 8 по 11
i 0		; Записываем целое число 0 в ячейки с 12 по 15
i 1		; Записываем целое число 1 в ячейки с 16 по 19
i 5		; Записываем целое число 5 в ячейки с 20 по 23
c 31 01 12	; Загружаем в первый регистр число, находящееся с 12 - 15 ячейку памяти (число - 0)
c 31 10 12	; Загружаем во второй регистр число, находящееся с 12 - 15 ячейку памяти (число - 0)
; Целочисленное сложение
c 2 01 4		; Сложение: 100 + 200 = 300 с сохранением в ячейках с 0 по 3
c 33 01 0		; Вывод результата
; Целочисленное вычитание
c 3 01 8		; Вычитание: 300 - 5 = 295 с сохранением в ячейках с 0 по 3
c 33 01 0		; Вывод результата
; Целочисленное умножение
c 4 11 4 8		; Умножение: 200 * 5 = 1000 с сохранением в ячейках с 4 по 7
c 33 01 4		; Вывод результата
; Целочисленное деление
c 5 11 4 8		; Деление: 1000 / 5 = 200 с сохранением в ячейках с 4 по 7
c 33 01 4		; Вывод результата
; Остаток от деления
c 6 01 4		; Остаток от деления: 295 % 200 = 95 с сохранением в ячейках с 0 по 3
c 33 01 0		; Вывод результата
; Инкремент
c 7 10 12		; Инкремент: ++0 = 1
c 33 01 12	; Вывод результата
; Декремент
c 8 10 12		; Декремент: --1 = 0
c 33 01 12	; Вывод результата
; Логическое И
c 9 11 16 20    	; Логическое: И 1 & 5 = 1 с сохранением в ячейках с 16 по 19
c 33 01 16	; Вывод результата
; Логическое ИЛИ
c 10 11 20 16    	; Логическое ИЛИ: 1 | 5 = 5 с сохранением в ячейках с 20 по 23
c 33 01 20	; Вывод результата
; Логическое НЕ
c 11 10 16    	; Логическое НЕ: ~1 = -2 с сохранением в ячейках с 16 по 19
c 33 01 16	; Вывод результата
; Исключающее ИЛИ
c 12 11 20 12	; Исключающее ИЛИ: 5 XOR 0 = 5 с сохранением в ячейках с 20 по 23
c 33 01 20	; Вывод результата
e 24                 	;  Устанавливаем IP на 24 ячейку памяти
